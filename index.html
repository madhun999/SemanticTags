    public void opAdd() {
       uint signbit = 0x8000; uint carrybit = 0x10000; uint mask = 0xFFFF;
       switch (this.bytesize) {
          case 1: signbit = 0x0080; carrybit = 0x0100; mask = 0x00FF; break;
          case 2: break; // Do nothing
          case 4: break; // TODO: Not supported yet
       }
       this.result = (uint)(this.operand1 + this.operand2 + (this.fCarry ? 1 : 0)); // Add with Carry
       this.fZero = ((result & mask) == 0); // Set Zero-result flag
       this.fParity = m_afParity[result & 0x00FF]; // Parity (which is for low-byte only) uses a lookuptable for speed
       this.fSign = ((result & signbit) != 0); // Sign/high bit set?
       this.fCarry = ((result & carrybit) != 0); // An unsigned overflow? Carry bit (High bit + 1) set?
       this.fAuxiliaryCarry = ((operand1 & 0x0010) != (result & 0x0010)); // Used for BCD operations

       // Signed overflow is a little more complicated:
       // Referring to this article: http://en.wikipedia.org/wiki/Overflow_flag :
       // "The overflow flag is set when the most significant bit (here considered the sign bit) is changed by adding
       // two numbers with the same sign (or subtracting two numbers with opposite signs). Overflow never occurs when
       // the sign of two addition operands are different (or the sign of two subtraction operands are the same)."
       // Thus we should get the following truth table:
       // (Sign/High Bit)
       // Operand 1: T T T T F F F F
       // Operand 2: T T F F T T F F
       // Result: T F T F T F T F
       // ------------------------------
       // Overflow: F T F F F F T F
       // // Using logical operators: Operand1 = Operand2 and Result != Operand1
       // this.fOverflow = (operand1 & 0x0080) == (operand2 & 0x0080) && (result & 0x0080) != (operand1 & 0x0080);
       // Or using bitwise operators: Operand1 XOR !Operand2 AND Result XOR Operand1
       this.fOverflow = (((operand1 ^ (~operand2)) & (operand1 ^ result)) & signbit) != 0;

       this.result &= mask; // Truncate result to requested #of bits.
    }
